package "枚举类" <<Rectangle>> {
    enum ActionEnum {
        +PUBLISH: 发布
        +CANCEL_PUBLISH: 取消发布
        +PAUSE: 暂停
        +RESUME: 恢复执行
        +START_WORK: 开始作业
        +FINISH_WORK: 完成作业
        +TERMINATE: 作废
        -code: String
        -desc: String
    }

    enum StateEnum {
        +CREATED: 创建
        +RELEASED: 发布
        +RUNNING: 执行
        +PAUSED: 暂停
        +COMPLETED: 完成
        +TERMINATED: 作废
        -code: String
        -desc: String
        +next(ActionEnum): StateEnum
    }
}

    class orderPermissionPolicy {
        -userMapper: userMapper
        +check(userId: Integer, action: ActionEnum): void
    }
    orderPermissionPolicy ..|> PermissionPolicy
    note right of orderPermissionPolicy
        PUBLISH / CANCEL_PUBLISH
        → 系统联动，不校验
        PAUSE / RESUME / TERMINATE
        → 需"生产主管"角色
        START_WORK / FINISH_WORK
        → 事实驱动，不校验
    end note

    class workOrderPermissionPolicy {
        -userMapper: userMapper
        +check(userId: Integer, action: ActionEnum): void
    }
    workOrderPermissionPolicy ..|> PermissionPolicy
    note right of workOrderPermissionPolicy
        PUBLISH / CANCEL_PUBLISH
        → 系统联动，不校验
        START_WORK / FINISH_WORK
        → 需本工单派工人员
        PAUSE / RESUME
        → 本工单人员或主管
        TERMINATE
        → 需"生产主管"角色
    end note

    class GatePolicy {
        -planMapper: planMapper
        -processFlowMapper: processFlowMapper
        -lineMapper: lineMapper
        -processMapper: processMapper
        -userMapper: userMapper
        +check(context: StateContext): void
        +checkOrder(context: StateContext, planNo: String, lineNo: String): void
    }
    GatePolicy ..|> GatePolicyInterface
    note right of GatePolicy
        check(): Plan 级门禁
          1. Process Gate: 工艺流程存在性
          2. Capacity Gate: 产线可用性
          3. Skill Gate: 工序人员技能
        checkOrder(): Order 级门禁
          1. Capacity Gate: 产线存在且空闲
          2. Process Gate: 产线绑定有效工艺流程
          3. Skill Gate: 工序人员技能
    end note
}

    class planStateServiceImpl {
        -planStateMachine: planStateMachine
        -permissionPolicy: planPermissionPolicy
        -gatePolicy: GatePolicy
        -auditService: AuditService
        -planMapper: planMapper
        -orderMapper: orderMapper
        -processFlowMapper: processFlowMapper
        -lineMapper: lineMapper
        -workOrderMapper: workOrderMapper
        +handle(planNo: String, action: ActionEnum, userId: Integer): void
        +handleLinkage(Plan, ActionEnum, Integer userId): void
        +syncPlanStatusFromOrders(planNo: String, userId: Integer): void
        -findLineByProcessFlowId(Integer): String
    }
    planStateServiceImpl ..|> planStateService
    note right of planStateServiceImpl
        handle(): 权限→状态机→门禁→
        状态更新→审计→联动
        handleLinkage():
          PUBLISH → 按产线拆分生成 Order
          PAUSE → 级联暂停 Order+WorkOrder
          RESUME → 级联恢复 Order+WorkOrder
          CANCEL_PUBLISH → 级联作废 Order+WorkOrder
          TERMINATE → 级联终止 Order+WorkOrder
        syncPlanStatusFromOrders():
          Order→Plan 上行聚合
          任一Order RUNNING → Plan RUNNING
          已执行Order全终态且≥1个COMPLETED
            → Plan COMPLETED（含TERMINATED）
          已执行Order全终态/暂停且≥1个PAUSED
            → Plan PAUSED（排除CREATED/RELEASED）
          完成时汇总COMPLETED订单产量到completedNum
    end note

    class orderServiceImpl {
        -orderMapper: orderMapper
        +getOrderByPlan(String): List<ProductionOrder>
        +getAllOrder(): Object
    }
    orderServiceImpl ..|> orderService

    orderStateServiceImpl ..|> orderStateService
    note right of orderStateServiceImpl
        handle(): 权限→状态机→门禁→
        状态更新→审计→联动
        handleLinkage():
          向上: syncPlanStatusFromOrders
          向下: PUBLISH → 生成+发布工单
                CANCEL_PUBLISH → 作废工单
                TERMINATE → 终止工单
                PAUSE → 暂停工单
                RESUME → 恢复工单
    end note

    class workOrderStateServiceImpl {

    workOrderStateServiceImpl ..|> workOrderStateService
    note right of workOrderStateServiceImpl
        handle(): 权限→状态机→
        员工归属校验→业务校验→
        差异化持久化→审计→联动
        handleLinkage():
          仅关键工单触发向上联动
          START_WORK → Order RUNNING
          FINISH_WORK → Order COMPLETED
          (汇总产量至 Order → Plan)
          PAUSE/RESUME/TERMINATE
          → 同步 Order 状态 → Plan
    end note
